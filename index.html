# Semantic Search — Embed query via LiteLLM, then search Qdrant
# pip install openai requests
from openai import OpenAI
import requests

# Step 1: Generate embedding
client = OpenAI(
    base_url="https://llm.aymenmastouri.io/v1",
    api_key="<your-api-key>"
)
embed = client.embeddings.create(
    model="nomic-embed-text",
    input=["What is a vector database?"]
)
query_vector = embed.data[0].embedding

# Step 2: Search Qdrant
response = requests.post(
    url="https://qdrant.aymenmastouri.io/collections/repo_docs/points/search",
    headers={"api-key": "<your-qdrant-api-key>"},
    json={"vector": query_vector, "limit": 5, "with_payload": True}
)

for r in response.json()["result"]:
    print(r["score"], r["payload"])
